-- 1. Users (with additional contact info)
CREATE TABLE users (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  email VARCHAR(255) NOT NULL UNIQUE,
  contact VARCHAR(20) NOT NULL
);

-- 2. Projects (with extended metadata)
CREATE TABLE projects (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  description TEXT,
  status VARCHAR(20) NOT NULL CHECK (status IN ('active', 'paused', 'completed', 'archived')),
  start_date TIMESTAMP NOT NULL,
  target_end_date TIMESTAMP,
  project_manager_id VARCHAR(36) NOT NULL,
  completion_percentage DECIMAL(5,2) DEFAULT 0.00,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  priority VARCHAR(20) CHECK (priority IN ('low', 'medium', 'high', 'critical')),
  client VARCHAR(50),
  department VARCHAR(50),
  FOREIGN KEY (project_manager_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 3. Teams (with creation timestamp)
CREATE TABLE teams (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  project_id VARCHAR(36) NOT NULL,
  team_lead_id VARCHAR(36) NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (team_lead_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 4. Team Members (with join timestamp)
CREATE TABLE team_members (
  user_id VARCHAR(36) NOT NULL,
  team_id VARCHAR(36) NOT NULL,
  joined_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (user_id, team_id),
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (team_id) REFERENCES teams(id) ON DELETE CASCADE
);

-- 5. Tasks (with due dates)
CREATE TABLE tasks (
  id VARCHAR(36) PRIMARY KEY,
  project_id VARCHAR(36) NOT NULL,
  team_id VARCHAR(36) NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  target_due_date TIMESTAMP,
  FOREIGN KEY (team_id) REFERENCES teams(id) ON DELETE CASCADE,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE
);

-- 6. Subtasks (with full task tracking)
CREATE TABLE subtasks (
  id VARCHAR(36) PRIMARY KEY,
  task_id VARCHAR(36) NOT NULL,
  description TEXT NOT NULL,
  priority VARCHAR(20) CHECK (priority IN ('low', 'medium', 'high', 'critical')),
  due_date TIMESTAMP,
  completed BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  assigned BOOLEAN NOT NULL DEFAULT FALSE,
  assigned_to VARCHAR(36) NOT NULL,
  project_id VARCHAR(36) NOT NULL,
  estimated_hours DECIMAL(5,2),
  parent_subtask_id VARCHAR(36),
  FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE,
  FOREIGN KEY (assigned_to) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (parent_subtask_id) REFERENCES subtasks(id) ON DELETE SET NULL
);

-- For tags (many-to-many relationship with subtasks)
CREATE TABLE tags (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(50) NOT NULL UNIQUE
);


-- 7. Milestones (project pipeline stages)
CREATE TABLE milestones (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  project_id VARCHAR(36) NOT NULL,
  description TEXT,
  due_date TIMESTAMP,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  sequence_no INTEGER NOT NULL CHECK (sequence_no > 0),
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE
);
